Problem: Find the maximum some of the subarray
This is a brute force technique
Time Complexity : O(N3)

Code : 
#include <iostream>
#include <climits>
using namespace std;

int main(){
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    int maxSub = INT_MIN;  
    for(int i = 0; i < n; i++){
        for(int j = i; j < n; j++){
            int sum = 0;
            for(int k = i; k <= j; k++){
                sum += a[k];
            }
            maxSub = max(maxSub, sum);
            
        }
    }
    cout<<maxSub;
    return 0;
}
